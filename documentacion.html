Descripción del Proyecto

COMEXCARE es un sistema empresarial desarrollado para Comex (empresa mexicana de recubrimientos) que permite la gestión remota de equipos de cómputo (agentes instalados en tiendas a nivel nacional), la distribución y recepción de archivos (actualizaciones de agente, archivos PVSI, DBF), y la generación de reportes de ventas avanzados con múltiples capas de cache y optimización.

El sistema soporta 500+ usuarios concurrentes con tiempos de respuesta <3s incluso procesando más de 7,000 registros, gracias a un sistema de cache escalonado (Redis + tablas materializadas + pre-agregación).

Ejemplo: AgentController - Método heartbeat

/**
 * Procesa el heartbeat de un agente.
 * Actualiza el estado online/offline y la última actividad.
 */
public function heartbeat(Request $request): JsonResponse
{
    $computer = Computer::where('short_key', $request->input('short_key'))
        ->firstOrFail();

    $computer->update([
        'last_heartbeat_at' => now(),
        'is_online' => true,
        'ip_address' => $request->ip(),
    ]);

    return response()->json(['status' => 'ok', 'time' => now()]);
}

Ejemplo: AgentController - Método register (Código de ejemplo)

public function register(Request $request): JsonResponse
{
    $validated = $request->validate([
        'computer_name' => 'required|string|max:255',
        'short_key' => 'required|string|max:50|unique:computers,short_key',
        'mac_address' => 'nullable|string|max:17',
        'group_short_key' => 'nullable|string|max:50',
    ]);

    $group = null;
    if ($request->filled('group_short_key')) {
        $group = GroupShortKey::where('short_key', $request->group_short_key)
            ->first()?->group_id;
    }

    $computer = Computer::create([
        'computer_name' => $validated['computer_name'],
        'short_key' => $validated['short_key'],
        'mac_address' => $validated['mac_address'] ?? null,
        'group_id' => $group,
        'ip_address' => $request->ip(),
        'agent_version' => $request->input('agent_version'),
        'is_online' => true,
        'last_heartbeat_at' => now(),
    ]);

    return response()->json($computer, 201);
}

Ejemplo: ReporteService (Capa de Optimización)

/**
 * ReportService - Servicio centralizado de reportes.
 * Procesa consultas SQL optimizadas con cache automático.
 */
class ReportService
{
    public function getReporteVendedores(
        array $params,
        bool $useCache = true
    ): Collection {
        $cacheKey = 'reporte.vendedores.' . md5(serialize($params));

        if ($useCache && Cache::has($cacheKey)) {
            return Cache::get($cacheKey);
        }

        $query = DB::table('vendedores_cache')
            ->select(
                'asesor',
                'nombre',
                DB::raw('SUM(venta) as total_venta'),
                DB::raw('COUNT(DISTINCT nota) as total_notas')
            )
            ->when($params['plaza'] ?? null,
                fn($q, $plaza) => $q->where('plaza', $plaza))
            ->groupBy('asesor', 'nombre')
            ->orderByDesc('total_venta');

        $results = $query->get();

        if ($useCache) {
            Cache::put($cacheKey, $results, now()->addMinutes(15));
        }

        return $results;
    }
}

Ejemplo: ReportService - Método de obtención de datos

// app/Services/ReportService.php (1026 líneas)
public function obtenerReporteVendedores(
    array $params,
    int $page = 1,
    int $perPage = 50
): array {
    $cacheKey = 'reporte:vendedores:' . md5(serialize($params));

    return Cache::remember($cacheKey, 3600, function () use ($params) {
        $query = VendedoresCache::query()
            ->select([
                'asesor', 'nombre',
                DB::raw('COALESCE(SUM(monto), 0) as total_venta'),
                DB::raw('COUNT(*) as total_notas'),
            ]);

        if (!empty($params['plaza'])) {
            $query->where('plaza', $params['plaza']);
        }
        if (!empty($params['tienda'])) {
            $query->where('tienda', $params['tienda']);
        }

        return $query
            ->groupBy('asesor', 'nombre')
            ->orderByDesc('total_venta')
            ->get();
    });
}

Ejemplo: ProcessDistributionJob

class ProcessDistributionJob implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

    public $timeout = 300;
    public $tries = 3;

    public function __construct(
        public Distribution $distribution
    ) {}

    public function handle(DistributionService $service): void
    {
        foreach ($this->distribution->targets as $target) {
            try {
                $service->sendDownloadCommand($target);
                $target->update(['status' => 'sending']);
            } catch (Exception $e) {
                $target->increment('attempts');
                $target->update([
                    'status' => 'failed',
                    'error_message' => $e->getMessage(),
                ]);
            }
        }
    }
}

class ApiRateLimiter
{
    public function handle(Request $request, Closure $next, string $type = 'default'): mixed
    {
        $limits = [
            'agents' => ['limit' => 30, 'window' => 60],
            'heartbeat' => ['limit' => 60, 'window' => 60],
            'default' => ['limit' => 100, 'window' => 60],
        ];

        $limit = $limits[$type] ?? $limits['default'];

        return RateLimiter::attempt(
            'api:' . $type . ':' . $request->ip(),
            $limit['limit'],
            function () use ($next, $request) {
                return $next($request);
            },
            $limit['window']
        ) ?: response()->json(['error' => 'Too many requests'], 429);
    }
}

Ejemplo: DistributionProgressUpdated Event

class DistributionProgressUpdated implements ShouldBroadcast
{
    use Dispatchable, InteractsWithSockets, SerializesModels;

    public function __construct(
        public Distribution $distribution,
        public int $completed,
        public int $total
    ) {}

    public function broadcastOn(): array
    {
        return [
            new PrivateChannel('distribution.' . $this->distribution->id),
            new Channel('distributions'),
        ];
    }

    public function broadcastWith(): array
    {
        return [
            'id' => $this->distribution->id,
            'completed' => $this->completed,
            'total' => $this->total,
            'percentage' => $this->total > 0
                ? round(($this->completed / $this->total) * 100, 2)
                : 0,
        ];
    }
}
